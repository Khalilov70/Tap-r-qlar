#include <iostream>
using namespace std;
int main() {
    cout << "1-dən 1000-ə qədər palindrom ədədlər:\n";
    for (int i=1;i<=1000;i++) {
        int n=i;
        int t=0;
        while (n>0) {
            int r=n%10;
            t=t*10+r;
            n/=10;
        }
        if (i==t)
            cout << i << " ";
    }
    cout << endl;
    return 0;
}
