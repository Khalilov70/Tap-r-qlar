#include <iostream>
using namespace std;
int main() {
    cout << "Rəqəmlərinin kvadratlarının cəmi özünə bərabər olan ədədlər (1 - 1000 aralığında):\n";
    for (int i=1;i<=1000;i++) {
        int n=i;
        int s=0;
        while (n>0) {
            int r=n%10; 
            s+=r*r;     
            n/=10;          
        }
        if (s==i)
            cout << i << " ";
    }
    cout << endl;
    return 0;
}
